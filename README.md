        function openCheckoutModal() {
            closeModal('cartModal');

        function closeModal(modalId) {
            document.getElementById(modalId).classList.remove('active');

        function changeQuantity(change) {
            productQuantity = Math.max(1, productQuantity + change);

        function addToCart(packageType, quantity = 1) {
            const packageData = {
                inicio: {
                    name: 'Pack Inicio',
                    price: 29.99,
                    image: 'https://sfile.chatglm.cn/images-ppt/bb42a792f1c0.jpg'

                existingItem.quantity += quantity;
            } else {
                cart.push({

                    ...packageData[packageType],
                    quantity: quantity
                });

            if (cart.length === 0) {
                cartContent.innerHTML = `
                    <div class="cart-empty">
                        <i class="material-icons">shopping_cart</i>
                        <p>Tu carrito está vacío</p>
                    </div>
                `;
                checkoutButton.style.display = 'none';
            } else {
                let cartHTML = '<div class="cart-items">';
                let total = 0;
                
                cart.forEach((item, index) => {
                    const itemTotal = item.price * item.quantity;
                    total += itemTotal;
                    
                    cartHTML += `
                        <div class="cart-item">
                            <img src="${item.image}" alt="${item.name}" class="cart-item-image">
                            <div class="cart-item-details">
                                <div class="cart-item-name">${item.name}</div>
                                <div class="cart-item-price">$${item.price.toFixed(2)}</div>
                            </div>
                            <div class="cart-item-quantity">
                                <button class="quantity-button" onclick="updateCartItemQuantity(${index}, -1)">
                                    <i class="material-icons">remove</i>
                                </button>
                                <div class="quantity-value">${item.quantity}</div>
                                <button class="quantity-button" onclick="updateCartItemQuantity(${index}, 1)">
                                    <i class="material-icons">add</i>
                                </button>
                                <button class="cart-item-remove" onclick="removeCartItem(${index})">
                                    <i class="material-icons">delete</i>
                                </button>
                            </div>
                        </div>
                    `;
                });
                
                cartHTML += '</div>';
                
                cartHTML += `
                    <div class="cart-summary">
                        <div class="cart-summary-row">
                            <span>Subtotal:</span>
                            <span>$${total.toFixed(2)}</span>
                        </div>
                        <div class="cart-summary-row">
                            <span>Envío:</span>
                            <span>$5.00</span>
                        </div>
                        <div class="cart-summary-total">
                            <span>Total:</span>
                            <span>$${(total + 5).toFixed(2)}</span>
                        </div>
                    </div>
                `;
                
                cartContent.innerHTML = cartHTML;
                checkoutButton.style.display = 'block';
            }
        }
        
        // Update cart item quantity
        function updateCartItemQuantity(index, change) {
            cart[index].quantity = Math.max(1, cart[index].quantity + change);
            updateCartCount();
            updateCartDisplay();
        }
        
        // Remove cart item
        function removeCartItem(index) {
            cart.splice(index, 1);
            updateCartCount();
            updateCartDisplay();
            showNotification('Producto eliminado del carrito', 'success');
        }
        
        // Checkout steps
        function nextCheckoutStep(currentStep) {
            // Validate current step
            if (currentStep === 1) {
                const fullName = document.getElementById('fullName').value;
                const email = document.getElementById('email').value;
                const phone = document.getElementById('phone').value;
                
                if (!fullName || !email || !phone) {
                    showNotification('Por favor completa todos los campos', 'error');
                    return;
                }
            } else if (currentStep === 2) {
                const address = document.getElementById('address').value;
                const city = document.getElementById('city').value;
                const zipCode = document.getElementById('zipCode').value;
                const country = document.getElementById('country').value;
                
                if (!address || !city || !zipCode || !country) {
                    showNotification('Por favor completa todos los campos', 'error');
                    return;
                }
            }
            
            // Update UI
            document.getElementById(`checkoutStep${currentStep}`).classList.remove('active');
            document.getElementById(`step${currentStep}`).classList.remove('active');
            document.getElementById(`step${currentStep + 1}`).classList.add('active');
            document.getElementById(`checkoutStep${currentStep + 1}`).classList.add('active');
        }
        
        function prevCheckoutStep(currentStep) {
            document.getElementById(`checkoutStep${currentStep}`).classList.remove('active');
            document.getElementById(`step${currentStep}`).classList.remove('active');
            document.getElementById(`step${currentStep - 1}`).classList.add('active');
            document.getElementById(`checkoutStep${currentStep - 1}`).classList.add('active');
        }
        
        // Select payment method
        function selectPaymentMethod(method) {
            selectedPaymentMethod = method;
            
            // Update UI
            document.getElementById('payuOption').classList.remove('selected');
            document.getElementById('mercadoPagoOption').classList.remove('selected');
            document.getElementById('payuForm').style.display = 'none';
            document.getElementById('mercadoPagoForm').style.display = 'none';
            
            if (method === 'payu') {
                document.getElementById('payuOption').classList.add('selected');
                document.getElementById('payuForm').style.display = 'block';
            } else if (method === 'mercadopago') {
                document.getElementById('mercadoPagoOption').classList.add('selected');
                document.getElementById('mercadoPagoForm').style.display = 'block';
            }
        }
        
        // Process payment
        function processPayment() {
            if (!selectedPaymentMethod) {
                showNotification('Por favor selecciona un método de pago', 'error');
                return;
            }
            
            // Simulate payment processing
            showNotification('Procesando pago...', 'success');
            
            setTimeout(() => {
                // Update order details
                const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0) + 5;
                document.getElementById('orderTotal').textContent = `$${total.toFixed(2)}`;
                document.getElementById('paymentMethod').textContent = selectedPaymentMethod === 'payu' ? 'PayU' : 'Mercado Pago';
                
                // Generate random order number
                const orderNumber = `VG-2023-${Math.floor(Math.random() * 9999)}`;
                document.getElementById('orderNumber').textContent = orderNumber;
                
                // Set current date
                const today = new Date();
                const dateStr = today.toLocaleDateString('es-ES', { 
                    day: 'numeric', 
                    month: 'long', 
                    year: 'numeric' 
                });
                document.getElementById('orderDate').textContent = dateStr;
                
                // Close checkout modal and open confirmation modal
                closeModal('checkoutModal');
                document.getElementById('confirmationModal').classList.add('active');
                
                // Clear cart
                cart = [];
                updateCartCount();
            }, 2000);
        }
        
        // Toggle FAQ
        function toggleFaq(element) {
            const faqItem = element.parentElement;
            faqItem.classList.toggle('active');
        }
        
        // Show notification
        function showNotification(message, type = 'success') {
            const notification = document.getElementById('notification');
            const notificationMessage = document.getElementById('notificationMessage');
            const notificationIcon = document.getElementById('notificationIcon');
            
            notificationMessage.textContent = message;
            notification.className = `notification ${type}`;
            
            if (type === 'success') {
                notificationIcon.textContent = 'check';
            } else if (type === 'error') {
                notificationIcon.textContent = 'error';
            }
            
            notification.classList.add('show');
            
            setTimeout(() => {
                notification.classList.remove('show');
            }, 3000);
        }
        
        // Initialize page
        document.addEventListener('DOMContentLoaded', function() {
            updateCartCount();
        });
